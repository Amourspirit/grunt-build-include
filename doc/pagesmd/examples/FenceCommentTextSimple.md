### Config

#### GruntFile.js

```js
module.exports = function (grunt) {
  grunt.initConfig({
    build_include: {
      default: {
        options: {
          match: {
            kind: 'bracketIncludeMulti',
            path: './fixtures/'
          },
          comment: {
            type: 'singleAsterisk'
          },
          text: {
            indent: true
          },
          fence: 'strict'
        },
        src: './src/interfaces.ts',
        dest: './scratch/js/interfaces.ts'
      }
    }
  });
  grunt.loadNpmTasks('grunt-build-include');
  grunt.registerTask('default', ['build_include:default']);
};
```
